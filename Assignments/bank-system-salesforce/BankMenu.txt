// -------- MENU DRIVEN BANK ACCOUNT SYSTEM --------

// Change choice value to test operations
Integer choice = 1;

// Customer Input Data
Integer customerId = 101;
String customerName = 'Krish Bansal';
String email = 'krish@gmail.com';
Date birthDate = Date.newInstance(2005, 1, 1);
String accountType = 'Savings';

// Menu Logic
switch on choice {
    
    // Add Customer
    when 1 {
        BankAccountSystem.addCustomer(
            customerId,
            customerName,
            email,
            birthDate,
            accountType
        );
    }

    // View Customers
    when 2 {
        BankAccountSystem.viewCustomers();
    }

    // Update Customer
    when 3 {
        BankAccountSystem.updateCustomer(
            customerId,
            'Current'
        );
    }

    // Delete Customer
    when 4 {
        BankAccountSystem.deleteCustomer(customerId);
    }

    when else {
        System.debug('Invalid Choice');
    }
}
