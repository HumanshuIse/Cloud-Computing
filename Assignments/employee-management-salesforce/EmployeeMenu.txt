// -------- MENU DRIVEN EMPLOYEE MANAGEMENT SYSTEM --------
// Change choice value to test different operations

Integer choice = 1;

// Employee Input Data
Integer empId = 101;
String empName = 'Krish Bansal';
String email = 'krish@gmail.com';
Date birthDate = Date.newInstance(2005, 1, 1);
String department = 'IT';

// Menu Driven Switch Case
switch on choice {
    
    // 1. Add Employee
    when 1 {
        EmployeeManagementSystem.addEmployee(
            empId,
            empName,
            email,
            birthDate,
            department
        );
    }

    // 2. View Employees
    when 2 {
        EmployeeManagementSystem.viewEmployees();
    }

    // 3. Update Employee Department
    when 3 {
        EmployeeManagementSystem.updateEmployee(
            empId,
            'Computer Engineering'
        );
    }

    // 4. Delete Employee
    when 4 {
        EmployeeManagementSystem.deleteEmployee(empId);
    }

    // Invalid Choice
    when else {
        System.debug('Invalid Choice');
    }
}
